function numberOfPairs(gloves){
    let pairs = 0
    let glovesPairs = gloves.reduce((acc, el)=> {
      acc[el] = (acc[el] || 0) + 1;
      return acc;
    }, {})
    for(color in glovesPairs){
      pairs += Math.floor(glovesPairs[color] / 2)
    }
    return pairs
}
